SONY.Global = (function(window, document) {

  'use strict';

  SONY.$window = $(window);
  SONY.$html = $(document.documentElement);
  SONY.$body = $(document.body);

  var self = {

    'init': function(config) {

      if ( window.enquire ){
        window.enquire.listen(100);
      }

      SONY.trigger('global:ready');

      log('SONY : Global : Initialized');
    }

  };

  return self;

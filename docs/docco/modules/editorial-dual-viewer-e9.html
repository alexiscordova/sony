As SonyDraggable returns scrubbing changes, update the top slide's width or height to match, and adjust the image container's width by that percentage's inverse to maintain the desired positioning.

 
    
    'onDrag': function(e) {
      var self = this;

      if ( self.axis == 'x' ) {

        self.$topSlide.css('width', (e.position.left) + '%');
        self.$topSlideImageContainer.css('width', 10000 / (e.position.left) + '%');

      } else if ( self.axis == 'y' ) {

        self.$topSlide.css('height', (e.position.top) + '%');
        self.$topSlideImageContainer.css('height', 10000 / (e.position.top) + '%');

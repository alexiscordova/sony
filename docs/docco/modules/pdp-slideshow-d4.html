    createThumbNav: function() {
      var self = this,
        $anchors = self.$thumbNav.find('a').hammer();

      $anchors.on('click', function(e) {
        e.preventDefault();

      $anchors.on('tap', function(e) {
        self.onThumbSelected(e);

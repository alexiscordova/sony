 /* TAB PLUGIN DEFINITION
  * ===================== */

  $.fn.tab = function ( option ) {
    var args = Array.prototype.slice.call( arguments, 1 );
    return this.each(function () {
      var $this = $(this),
          tab = $this.data('tab');

      if ( !tab ) {
        tab = new Tab(this);
        $this.data( 'tab', tab );
      }

      if ( typeof option === 'string' ) {
        tab[ option ].apply( tab, args );
      }
    });
  };

  $.fn.tab.Constructor = Tab;

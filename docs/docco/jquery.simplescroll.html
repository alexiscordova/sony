Convenience method which attaches a click event to the collection which scrolls to the defined target (via a function) or the href attribute

 
  $.fn.simplescroll = function( options, fn ) {
    return this.each(function() {
      $(this).on('click', function(evt) {
        evt.preventDefault();
        options = options || {};
        options.target = $.isFunction(options.target) ? options.target.call(this) : this.getAttribute('href');
        $.simplescroll( options, fn );
      });
    });
  };

  $.simplescroll.options = {
    target: 'body',
    speed: 600,
    easing: $.easing.easeOutQuad ? 'easeOutQuad' : 'swing',
    showHash: false,
    callback: $.noop,
    offset: 0
  };
